function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  var data = JSON.parse(e.postData.contents);

  sheet.appendRow([
    new Date(),
    data.userAgent || "brak",
    data.page || "brak",
    data.referrer || "brak",
    data.screen || "brak",
    data.language || "brak",
    data.timezone || "brak",
    data.time || "brak"
  ]);

  return ContentService.createTextOutput("OK").setMimeType(ContentService.MimeType.TEXT);
}
